class SEBlock(nn.Module):
    def __init__(self, in_channels, reduction=8):
        super(SEBlock, self).__init__()
        self.fc1 = nn.Linear(in_channels, in_channels // reduction)
        self.fc2 = nn.Linear(in_channels // reduction, in_channels)

    def forward(self, x):
        batch, channels, height, width = x.size()
        se = F.adaptive_avg_pool2d(x, (1, 1)).view(batch, channels)
        se = F.relu(self.fc1(se))
        se = torch.sigmoid(self.fc2(se))
        se = se.view(batch, channels, 1, 1)
        return x * se


class MMAMBlock(nn.Module):
    def __init__(self, in_channels, reduction=8):
        super(MMAMBlock, self).__init__()

        self.conv_fg = nn.Conv2d(in_channels, in_channels, kernel_size=3, padding=1, groups=in_channels)  
        self.conv_bg = nn.Conv2d(in_channels, in_channels, kernel_size=3, padding=1, groups=in_channels)  
        self.conv_edge = nn.Conv2d(in_channels, in_channels, kernel_size=3, padding=1, groups=in_channels)  

        self.se_fg = SEBlock(in_channels, reduction=reduction)
        self.se_bg = SEBlock(in_channels, reduction=reduction)
        self.se_edge = SEBlock(in_channels, reduction=reduction)

        self.final_conv = nn.Conv2d(in_channels * 3, in_channels, kernel_size=1)

    def forward(self, x):
       
        fg_features = self.conv_fg(x)
        bg_features = self.conv_bg(x)
        edge_features = self.conv_edge(x)

        fg_features = self.se_fg(fg_features)
        bg_features = self.se_bg(bg_features)
        edge_features = self.se_edge(edge_features)

        combined = torch.cat([fg_features, bg_features, edge_features], dim=1)
        output = self.final_conv(combined)
        return output

class ChannelAttention(nn.Module):
    def __init__(self, channels, reduction=8):
        super(ChannelAttention, self).__init__()
        self.fc1 = nn.Conv2d(channels, channels // reduction, kernel_size=1)
        self.fc2 = nn.Conv2d(channels // reduction, channels, kernel_size=1)

    def forward(self, x):
        avg_out = torch.mean(x, dim=(2, 3), keepdim=True)
        max_out, _ = torch.max(x, dim=(2, 3), keepdim=True)
        out = self.fc1(avg_out) + self.fc1(max_out)
        out = F.relu(out)
        out = self.fc2(out)
        return torch.sigmoid(out) * x


class SpatialAttention(nn.Module):
    def __init__(self, kernel_size=7):
        super(SpatialAttention, self).__init__()
        self.conv = nn.Conv2d(2, 1, kernel_size=kernel_size, padding=kernel_size // 2)

    def forward(self, x):
        avg_out = torch.mean(x, dim=1, keepdim=True)
        max_out, _ = torch.max(x, dim=1, keepdim=True)
        out = torch.cat([avg_out, max_out], dim=1)
        out = self.conv(out)
        return torch.sigmoid(out) * x


class ShapeAwareModule(nn.Module):
    def __init__(self, channels):
        super(ShapeAwareModule, self).__init__()
        self.conv3x3 = nn.Sequential(
            nn.Conv2d(channels, channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(channels),nn.ReLU(inplace=True))
        
        self.channel_attention = ChannelAttention(channels)
        self.spatial_attention = SpatialAttention()

    def forward(self, e1, e2, e3):
        
        e1_2 = self.channel_attention(torch.abs(self.conv3x3(e1) - self.conv3x3(e2)))
        e2_3 = self.channel_attention(torch.abs(self.conv3x3(e2) - self.conv3x3(e3)))
        
        e1_3 = torch.abs(e1_2 - e2_3)  
        e1_2_3 = e1_2 + e1_3  

        combined_features = torch.cat([e1_2_3, e2_3], dim=1)
        output = self.spatial_attention(self.conv3x3(combined_features))
        return output

class ChannelAttention(nn.Module):
    def __init__(self, channels, reduction=8):
        super(ChannelAttention, self).__init__()
        self.fc1 = nn.Conv2d(channels, channels // reduction, kernel_size=1)
        self.fc2 = nn.Conv2d(channels // reduction, channels, kernel_size=1)

    def forward(self, x):
        avg_out = torch.mean(x, dim=(2, 3), keepdim=True)
        max_out = torch.amax(x, dim=(2, 3), keepdim=True)  
        out = self.fc1(avg_out) + self.fc1(max_out)
        out = F.relu(out)
        out = self.fc2(out)
        return torch.sigmoid(out) * x


class SpatialAttention(nn.Module):
    def __init__(self, kernel_size=7):
        super(SpatialAttention, self).__init__()
        self.conv = nn.Conv2d(2, 1, kernel_size=kernel_size, padding=kernel_size // 2)

    def forward(self, x):
        avg_out = torch.mean(x, dim=1, keepdim=True)
        max_out, _ = torch.max(x, dim=1, keepdim=True)
        out = torch.cat([avg_out, max_out], dim=1)
        out = self.conv(out)
        return torch.sigmoid(out) * x


class CrossSemanticAttention(nn.Module):
    def __init__(self, channels, reduction=8):
        super(CrossSemanticAttention, self).__init__()
        self.channel_attention = ChannelAttention(channels, reduction)
        self.spatial_attention = SpatialAttention()

    def forward(self, x):
        x = self.channel_attention(x)
        x = self.spatial_attention(x)
        return x


class NonLocalBlock(nn.Module):
    def __init__(self, channels):
        super(NonLocalBlock, self).__init__()
        self.theta = nn.Conv2d(channels, channels // 2, kernel_size=1)
        self.phi = nn.Conv2d(channels, channels // 2, kernel_size=1)
        self.g = nn.Conv2d(channels, channels // 2, kernel_size=1)
        self.out_conv = nn.Conv2d(channels // 2, channels, kernel_size=1)

    def forward(self, x):
        batch_size, _, h, w = x.size()
        theta_x = self.theta(x).view(batch_size, -1, h * w)       
        phi_x = self.phi(x).view(batch_size, -1, h * w)           
        g_x = self.g(x).view(batch_size, -1, h * w)               

        attention = torch.matmul(theta_x.transpose(1, 2), phi_x)  
        attention = F.softmax(attention, dim=-1)

        out = torch.matmul(attention, g_x.transpose(1, 2)).transpose(1, 2)
        out = out.view(batch_size, -1, h, w)
        out = self.out_conv(out)
        return out + x


class FeatureEnhancementModule(nn.Module):
    def __init__(self, in_channels, out_channels):
        super(FeatureEnhancementModule, self).__init__()
        
        self.conv_dilated1 = nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1, dilation=1)
        self.conv_dilated3 = nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=3, dilation=3)
        self.conv_dilated5 = nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=5, dilation=5)
  
        
        self.csa1 = CrossSemanticAttention(out_channels)
        self.csa2 = CrossSemanticAttention(out_channels)
        self.csa3 = CrossSemanticAttention(out_channels)
        
        self.non_local = NonLocalBlock(in_channels)
        
        self.final_conv = nn.Conv2d(out_channels * 3, out_channels, kernel_size=3, padding=1)

    def forward(self, x):
        out1 = self.csa1(self.conv_dilated1(x))
        out2 = self.csa2(self.conv_dilated3(x))
        out3 = self.csa3(self.conv_dilated5(x))
        
        multi_scale_features = torch.cat([out1, out2, out3], dim=1)
        out = self.final_conv(multi_scale_features)
        
        residual = self.non_local(x)
        out += residual
        return out

class FinalSegmentationModel(nn.Module):
    def __init__(self, num_classes=1):
        super(FinalSegmentationModel, self).__init__()

        # Load PVTv2 model with features only
        self.PVTV2 = timm.create_model('pvt_v2_b1', pretrained=True, features_only=True)
        for param in self.PVTV2.parameters():
            param.requires_grad = False

        # Load EfficientNet-B0 with features only
        self.efficientnet = timm.create_model('efficientnet_b0', pretrained=True, features_only=True)


        self.conv1x1_pvt56 = nn.Conv2d(64, 96, kernel_size=1)
        self.conv1x1_pvt28 = nn.Conv2d(128, 128,kernel_size=1)
        self.conv1x1_pvt14 = nn.Conv2d(320, 192, kernel_size=1)
        self.conv1x1_pvt7 = nn.Conv2d(512, 256, kernel_size=1)

        self.conv1x1_eff56 = nn.Conv2d(24, 96, kernel_size=1)
        self.conv1x1_eff28 = nn.Conv2d(40, 128, kernel_size=1)
        self.conv1x1_eff14 = nn.Conv2d(112, 192, kernel_size=1)
        self.conv1x1_eff7 = nn.Conv2d(320, 256, kernel_size=1)

        # Define 1x1 convolution layers to reduce channels after concatenation
        self.reduce_channels56 = nn.Conv2d(96 + 96, 96, kernel_size=3,padding=1)  
        self.reduce_channels28 = nn.Conv2d(128 + 128, 128,kernel_size=3,padding=1)  
        self.reduce_channels14 = nn.Conv2d(192 + 192, 192, kernel_size=3,padding=1)  
        self.reduce_channels7 = nn.Conv2d(256 + 256, 256, kernel_size=3,padding=1)  

        self.reduce_fusion14 = nn.Conv2d(192 + 256, 192, kernel_size=1)
        self.reduce_fusion28 = nn.Conv2d(128 + 192, 128, kernel_size=1)
        self.reduce_fusion56 = nn.Conv2d(96 + 128, 96,kernel_size=1)

        # FEM modules for each resolution
        self.fem_28 = FeatureEnhancementModule(in_channels=128, out_channels=128)
        self.fem_14 = FeatureEnhancementModule(in_channels=192, out_channels=192)
        self.fem_7 = FeatureEnhancementModule(in_channels=256, out_channels=256)

        # SAM and attention layers
        self.conv3x3_e1 = nn.Conv2d(96, 192, kernel_size=1)
        self.conv3x3_e2 = nn.Conv2d(128, 192, kernel_size=1)
        self.conv3x3_e3 = nn.Conv2d(192, 192, kernel_size=1)

        self.channel_attention_e1_2 = ChannelAttention(192)
        self.channel_attention_e2_3 = ChannelAttention(192)
        self.spatial_attention = SpatialAttention()

        # Decoder layers
        self.upsample4 = nn.Upsample(scale_factor=2, mode='bilinear', align_corners=True)
        self.conv_decoder4 = nn.Sequential(nn.Conv2d(352, 192, kernel_size=1), nn.BatchNorm2d(192), nn.GELU())
        self.mmam4 = MMAMBlock(in_channels=352)

        self.upsample3 = nn.Upsample(scale_factor=2, mode='bilinear', align_corners=True)
        self.conv_decoder3 = nn.Sequential(nn.Conv2d(320, 256, kernel_size=1), nn.BatchNorm2d(256), nn.GELU())
        self.mmam3 = MMAMBlock(in_channels=320)

        self.upsample2 = nn.Upsample(scale_factor=2, mode='bilinear', align_corners=True)
        self.conv_decoder2 = nn.Sequential(nn.Conv2d(352, 192, kernel_size=1), nn.BatchNorm2d(192), nn.GELU())
        self.mmam2 = MMAMBlock(in_channels=352)

        self.upsample1 = nn.Upsample(scale_factor=2, mode='bilinear', align_corners=True)
        self.conv_decoder1 = nn.Sequential(nn.Conv2d(192, 96, kernel_size=1), nn.BatchNorm2d(96), nn.GELU())
        self.mmam1 = MMAMBlock(in_channels=192)

        self.upsample0 = nn.Upsample(scale_factor=2, mode='bilinear', align_corners=True)
        self.conv_decoder0 = nn.Sequential(nn.Conv2d(96, 96, kernel_size=1), nn.BatchNorm2d(96), nn.GELU())


        self.final_conv = nn.Conv2d(96, num_classes, kernel_size=1)

    def forward(self, x):
        # Get features from both PVTv2 and EfficientNet-B0
        pvt_features = self.PVTV2(x)
        eff_features = self.efficientnet(x)

        # Extract features for each resolution
        pvt_features56 = self.conv1x1_pvt56(pvt_features[-4])
        pvt_features28 = self.conv1x1_pvt28(pvt_features[-3])
        pvt_features14 = self.conv1x1_pvt14(pvt_features[-2])
        pvt_features7 = self.conv1x1_pvt7(pvt_features[-1])

        eff_features56 = self.conv1x1_eff56(eff_features[-4])
        eff_features28 = self.conv1x1_eff28(eff_features[-3])
        eff_features14 = self.conv1x1_eff14(eff_features[-2])
        eff_features7 = self.conv1x1_eff7(eff_features[-1])

        # Combine features from both backbones
        combined56 = self.reduce_channels56(torch.cat([pvt_features56, eff_features56], dim=1))
        combined28 = self.reduce_channels28(torch.cat([pvt_features28, eff_features28], dim=1))
        combined14 = self.reduce_channels14(torch.cat([pvt_features14, eff_features14], dim=1))
        combined7 = self.reduce_channels7(torch.cat([pvt_features7, eff_features7], dim=1))

        upsampled7 = F.interpolate(combined7, size=combined14.shape[2:], mode='bilinear', align_corners=True)
        combined14 = self.reduce_fusion14(torch.cat([combined14, upsampled7], dim=1))

        upsampled14 = F.interpolate(combined14, size=combined28.shape[2:], mode='bilinear', align_corners=True)
        combined28 = self.reduce_fusion28(torch.cat([combined28, upsampled14], dim=1))

        upsampled28 = F.interpolate(combined28, size=combined56.shape[2:], mode='bilinear', align_corners=True)
        combined56 = self.reduce_fusion56(torch.cat([combined56, upsampled28], dim=1))

        d4 = self.upsample4(combined7)
        sam_output_upsampled = F.interpolate(combined56, size=d4.shape[2:], mode='bilinear', align_corners=True)
        d4 = torch.cat([d4, sam_output_upsampled], dim=1)
        d4 = self.mmam4(d4)
        d4 = self.conv_decoder4(d4)

        d3 = self.upsample3(d4)
        combined28_upsampled = F.interpolate(combined28, size=d3.shape[2:], mode='bilinear', align_corners=True)
        d3 = torch.cat([d3, combined28_upsampled], dim=1)
        d3 = self.mmam3(d3)
        d3 = self.conv_decoder3(d3)

        d2 = self.upsample2(d3)
        combined56_upsampled = F.interpolate(combined56, size=d2.shape[2:], mode='bilinear', align_corners=True)
        d2 = torch.cat([d2, combined56_upsampled], dim=1)
        d2 = self.mmam2(d2)
        d2 = self.conv_decoder2(d2)

        d1 = self.upsample1(d2)
        d1 = self.mmam1(d1)
        d1 = self.conv_decoder1(d1)


        d0 = self.upsample0(d1)
        d0 = self.conv_decoder0(d0)
        return self.final_conv(d0)
